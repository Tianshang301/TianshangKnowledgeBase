# x86_64汇编与逆向

## 一、x86_64架构概述

x86_64（AMD64/Intel 64）是当前主流的桌面和服务器CPU架构。

寄存器体系：

| 寄存器 | 64位 | 32位 | 16位 | 8位 | 用途 |
|--------|------|------|------|------|------|
| 累加器 | RAX | EAX | AX | AL | 运算结果 |
| 基址 | RBX | EBX | BX | BL | 通用 |
| 计数器 | RCX | ECX | CX | CL | 循环计数 |
| 数据 | RDX | EDX | DX | DL | 乘除扩展 |
| 源变址 | RSI | ESI | SI | SIL | 字符串源 |
| 目标变址 | RDI | EDI | DI | DIL | 字符串目标 |
| 基址指针 | RBP | EBP | BP | BPL | 栈帧基址 |
| 栈指针 | RSP | ESP | SP | SPL | 栈顶指针 |
| 通用 | R8-R15 | R8d-R15d | R8w-R15w | R8b-R15b | 扩展通用 |

指令指针：RIP（64位指令指针）

## 二、基本指令集

数据传送指令：

```assembly
mov rax, rbx        ; rax = rbx
mov rax, [rbx]      ; rax = *rbx (内存读取)
mov [rbx], rax      ; *rbx = rax (内存写入)
movzx rax, byte [rsi]  ; 零扩展加载
movsx rax, byte [rsi]  ; 符号扩展加载
lea rax, [rbx+rcx*4]   ; 计算地址 (不访问内存)
```

算术指令：

```assembly
add rax, rbx        ; rax = rax + rbx
sub rax, rbx        ; rax = rax - rbx
imul rax, rbx       ; rax = rax * rbx (有符号)
mul rbx             ; rdx:rax = rax * rbx (无符号)
div rbx             ; rax = rdx:rax / rbx, rdx = 余数
inc rax             ; rax++
dec rax             ; rax--
```

控制流：

```assembly
cmp rax, rbx        ; 比较，设置标志位
je label            ; jump if equal (ZF=1)
jne label           ; jump if not equal
jg label            ; jump if greater (有符号)
jl label            ; jump if less (有符号)
jge label           ; jump if greater or equal
jle label           ; jump if less or equal
ja  label           ; jump if above (无符号)
jb  label           ; jump if below (无符号)

jmp label           ; 无条件跳转
call func           ; 调用函数 (push rip, jmp func)
ret                 ; 返回 (pop rip)
```

## 三、函数调用约定

System V AMD64 ABI（Linux/macOS）：

```
整数/指针参数: RDI, RSI, RDX, RCX, R8, R9 (从左到右)
浮点参数:      XMM0-XMM7
返回值:        RAX (整数), XMM0 (浮点)
栈对齐:       RSP在call前必须是16字节对齐
保留寄存器:   RBX, RBP, R12-R15 (被调用者保存)
```

```assembly
; 函数: int add_and_multiply(int a, int b, int c)
; 返回: (a + b) * c
add_and_multiply:
    ; prologue
    push rbp
    mov rbp, rsp

    ; a = edi, b = esi, c = edx
    mov eax, edi
    add eax, esi        ; eax = a + b
    imul eax, edx       ; eax = (a + b) * c

    ; epilogue
    pop rbp
    ret

; 调用
main:
    mov edi, 10         ; 参数1: a = 10
    mov esi, 20         ; 参数2: b = 20
    mov edx, 3          ; 参数3: c = 3
    call add_and_multiply
    ; eax = (10 + 20) * 3 = 90
```

## 四、系统调用

Linux x86_64系统调用通过`syscall`指令实现：

```
参数: RDI, RSI, RDX, R10, R8, R9
系统调用号: RAX
触发: syscall
返回值: RAX（负数为错误码）
```

```assembly
; 向 stdout 写入字符串
section .data
    msg db "Hello, World!", 10   ; 10 = newline
    len equ $ - msg

section .text
    global _start

_start:
    ; sys_write(1, msg, len)
    mov rax, 1          ; syscall number: sys_write
    mov rdi, 1          ; fd: stdout
    lea rsi, [msg]      ; buf: 字符串地址
    mov rdx, len        ; count: 字符串长度
    syscall

    ; sys_exit(0)
    mov rax, 60         ; syscall number: sys_exit
    xor rdi, rdi        ; exit code: 0
    syscall
```

常用系统调用号：

| 系统调用 | 编号 | 参数1 (rdi) | 参数2 (rsi) | 参数3 (rdx) |
|----------|------|-------------|-------------|-------------|
| sys_read | 0 | fd | buf | count |
| sys_write | 1 | fd | buf | count |
| sys_open | 2 | pathname | flags | mode |
| sys_close | 3 | fd | - | - |
| sys_mmap | 9 | addr | length | prot |
| sys_exit_group | 231 | exit_code | - | - |

## 五、逆向工程基础

ELF文件结构：

```
ELF Header
    Program Headers (段)
    Section Headers (节)
        .text: 代码段
        .data: 已初始化数据
        .bss:  未初始化数据
        .rodata: 只读数据
        .plt/.got: 动态链接
```

常用逆向工具：

| 工具 | 类型 | 适用平台 | 功能 |

| IDA Pro | 反汇编器 | 跨平台 | 商业，最强反汇编 |
| x64dbg | 动态调试器 | Windows | 用户态调试 |
| GDB | 调试器 | Linux | 命令行调试 |

## 七、常见逆向场景

反调试检测与绕过：

```assembly
; ptrace检测
ptrace_detect:
    mov rax, 101        ; sys_ptrace
    xor rdi, rdi        ; PTRACE_TRACEME = 0
    xor rsi, rsi
    xor rdx, rdx
    syscall
    test rax, rax
    js   being_debugged  ; 负数表示被跟踪

; /proc/self/status检测
check_tracer_pid:
    open "/proc/self/status"
    read file
    search "TracerPid:"
    if value != 0:
        being_debugged
```

字符串混淆：

```python
# 对抗静态分析的字符串加密
def xor_string(data, key=0xAA):
    return bytes(b ^ key for b in data)

# 生成混淆字符串
original = "sensitive_data"
obfuscated = xor_string(original.encode())
print(f"Obfuscated: {obfuscated.hex()}")
```

## 八、链接与加载

```assembly
; 动态链接示例 - 调用libc函数
extern printf
extern exit

section .data
    fmt db "Result: %d", 10, 0

section .text
    global main

main:
    push rbp
    mov rbp, rsp

    mov rdi, fmt
    mov rsi, 42
    xor rax, rax        ; 无浮点参数
    call printf

    xor rdi, rdi
    call exit
```

```bash
# 编译和链接
nasm -f elf64 program.asm -o program.o
gcc -no-pie program.o -o program

# 查看动态链接信息
ldd program
readelf -d program

# 查看PLT/GOT
objdump -d -j .plt program
```

## 九、Shellcode编写

```assembly
; execve("/bin/sh", NULL, NULL) shellcode
section .text
    global _start

_start:
    ; 清除寄存器
    xor rdx, rdx        ; envp = NULL
    xor rsi, rsi        ; argv = NULL

    ; 构建 "/bin/sh" 字符串
    push rdx            ; 字符串结尾 null
    mov rax, 0x68732f2f6e69622f  ; "/bin//sh"
    push rax

    mov rdi, rsp        ; pathname = "/bin//sh"

    mov rax, 59         ; sys_execve
    syscall

    ; sys_exit
    mov rax, 60
    xor rdi, rdi
    syscall
```

Shellcode提取：

```bash
# 编译和提取shellcode
nasm -f bin shellcode.asm -o shellcode.bin
xxd -i shellcode.bin | head -5

# 测试shellcode
unsigned char code[] = {
    0x48, 0x31, 0xd2, 0x48, 0x31, 0xf6, ...
};
```
