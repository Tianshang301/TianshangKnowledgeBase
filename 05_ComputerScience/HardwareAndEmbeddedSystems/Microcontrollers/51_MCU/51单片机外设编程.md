# 51单片机外设编程

## 一、51单片机架构概述

8051内核单片机采用Harvard架构，程序存储器和数据存储器分离。

内部资源：

| 资源 | 标准8051 | 增强型（如STC89C52） |
|------|----------|---------------------|
| 程序存储器 | 4KB ROM | 8KB Flash |
| 数据存储器 | 128B RAM | 256B RAM |
| 定时器/计数器 | 2个16位 | 3个16位 |
| 中断源 | 5个 | 8个 |
| IO口 | 4个8位（P0-P3） | 4个8位（P0-P3） |
| 串口 | 1个UART | 1个UART |
| 工作频率 | 0-24MHz | 0-40MHz |

特殊功能寄存器（SFR）是51单片机的控制核心，分布在0x80-0xFF地址空间。

## 二、GPIO口编程

标准的51单片机IO口为准双向口，内部有上拉电阻（P0口除外）。

LED流水灯示例：

```c
#include <reg52.h>

#define LED_PORT P1

void delay_ms(unsigned int ms) {
    unsigned int i, j;
    for (i = 0; i < ms; i++)
        for (j = 0; j < 123; j++);
}

void main() {
    unsigned char led_val = 0xFE;  // 1111 1110

    while (1) {
        LED_PORT = led_val;
        delay_ms(200);
        led_val = (led_val << 1) | 0x01;  // 左移流水
        if (led_val == 0xFF)
            led_val = 0xFE;
    }
}
```

独立按键输入检测：

```c
sbit KEY = P3^2;
sbit LED = P1^0;

void delay_ms(unsigned int ms) {
    unsigned int i, j;
    for (i = 0; i < ms; i++)
        for (j = 0; j < 123; j++);
}

void main() {
    LED = 1;
    while (1) {
        if (KEY == 0) {          // 检测按下
            delay_ms(10);        // 消抖
            if (KEY == 0) {
                LED = ~LED;      // 翻转LED
                while (KEY == 0); // 等待释放
            }
        }
    }
}
```

## 三、定时器/计数器

定时器工作模式：

| 模式 | 位数 | 计数器结构 | 自动重装 |
|------|------|-----------|----------|
| 模式0 | 13位 | TL1低5位 + TH1高8位 | 否 |
| 模式1 | 16位 | TL1 + TH1 | 否 |
| 模式2 | 8位 | TL1（自动重装到TH1）| 是 |
| 模式3 | 8位+8位 | 两个独立的8位计数器（仅T0）| 否 |

定时器初值计算：

$$
TH = \frac{65536 - T_{target} \times f_{osc} / 12}{256}
$$

$$
TL = (65536 - T_{target} \times f_{osc} / 12) \mod 256
$$

定时器模式1（16位）示例：

```c
#include <reg52.h>

sbit LED = P1^0;

void timer0_init() {
    TMOD &= 0xF0;          // 清除T0设置
    TMOD |= 0x01;          // T0: 模式1, 16位定时器

    // 定时50ms @ 12MHz
    TH0 = 0x4C;            // 高8位
    TL0 = 0x00;            // 低8位

    ET0 = 1;               // 使能T0中断
    EA = 1;                // 使能全局中断
    TR0 = 1;               // 启动定时器
}

unsigned char count = 0;

void timer0_isr() interrupt 1 {
    TH0 = 0x4C;            // 重装初值
    TL0 = 0x00;

    count++;
    if (count >= 20) {     // 20*50ms = 1s
        count = 0;
        LED = ~LED;
    }
}

void main() {
    timer0_init();
    while (1);
}
```

## 四、外部中断

51单片机有两个外部中断源：INT0 (P3^2) 和 INT1 (P3^3)。

```c
#include <reg52.h>

sbit KEY = P3^2;  // INT0

void delay_ms(unsigned int ms) {
    unsigned int i, j;
    for (i = 0; i < ms; i++)
        for (j = 0; j < 123; j++);
}

void ext0_init() {
    IT0 = 1;               // 下降沿触发
    EX0 = 1;               // 使能INT0中断
    EA = 1;                // 使能全局中断
}

void ext0_isr() interrupt 0 {
    // 中断服务程序
    delay_ms(10);          // 硬件消抖
    if (KEY == 0) {
        P1 = ~P1;          // P1口取反
    }
}

void main() {
    ext0_init();
    P1 = 0xFF;
    while (1);
}
```

中断优先级：

| 优先级 | 中断源 | 自然优先级 |
|--------|--------|-----------|
| 最高 | INT0 (IE0) | 1 |
| | TF0 (定时器0) | 2 |
| | INT1 (IE1) | 3 |
| | TF1 (定时器1) | 4 |
| 最低 | TI/RI (串口) | 5 |

## 五、串口通信

UART工作在模式1（8位可变波特率）最为常用。

波特率计算：

$$
SMOD = 0: \quad BaudRate = \frac{f_{osc}}{32 \times 12 \times (256 - TH1)}
$$

$$
SMOD = 1: \quad BaudRate = \frac{f_{osc}}{16 \times 12 \times (256 - TH1)}
$$

```c
#include <reg52.h>

void uart_init() {
    SCON = 0x50;           // 模式1, 8位UART, 允许接收
    TMOD &= 0x0F;
    TMOD |= 0x20;          // T1: 模式2, 8位自动重装

    // 9600波特率 @ 11.0592MHz
    TH1 = 0xFD;
    TL1 = 0xFD;

    TR1 = 1;               // 启动定时器1
    ES = 1;                // 使能串口中断
    EA = 1;                // 使能全局中断
}

void uart_send_byte(unsigned char dat) {
    SBUF = dat;
    while (!TI);           // 等待发送完成
    TI = 0;
}

void uart_send_string(unsigned char *str) {
    while (*str) {
        uart_send_byte(*str++);
    }
}

void uart_isr() interrupt 4 {
    unsigned char recv;

    if (RI) {              // 接收中断
        RI = 0;
        recv = SBUF;
        uart_send_byte(recv);  // 回显
    }
    if (TI) {              // 发送中断
        TI = 0;
    }
}

void main() {
    uart_init();
    uart_send_string("51 UART Ready\r\n");
    while (1);
}
```

## 六、I2C总线通信

使用IO口模拟I2C协议：

```c
sbit SDA = P2^0;
sbit SCL = P2^1;

void i2c_delay() {
    _nop_(); _nop_();
}

void i2c_start() {
    SDA = 1;
    SCL = 1;
    i2c_delay();
    SDA = 0;
    i2c_delay();
    SCL = 0;
}

void i2c_stop() {
    SDA = 0;
    SCL = 1;
    i2c_delay();
    SDA = 1;
    i2c_delay();
}

bit i2c_write_byte(unsigned char dat) {
    unsigned char i;
    for (i = 0; i < 8; i++) {
        SDA = (dat & 0x80) ? 1 : 0;
        dat <<= 1;
        SCL = 1;
        i2c_delay();
        SCL = 0;
    }
    SDA = 1;               // 释放总线
    SCL = 1;
    i2c_delay();
    bit ack = SDA;         // 读取应答
    SCL = 0;
    return ack;
}

unsigned char i2c_read_byte(bit ack) {
    unsigned char i, dat = 0;
    SDA = 1;
    for (i = 0; i < 8; i++) {
        dat <<= 1;
        SCL = 1;
        i2c_delay();
        if (SDA) dat |= 0x01;
        SCL = 0;
        i2c_delay();
    }
    SDA = ack ? 0 : 1;    // 发送应答
    SCL = 1;
    i2c_delay();
    SCL = 0;
    SDA = 1;
    return dat;
}
```

## 七、LCD1602液晶驱动

```c
#define LCD_DATA P0
sbit LCD_RS = P2^5;
sbit LCD_RW = P2^6;
sbit LCD_EN = P2^7;

void lcd_write_cmd(unsigned char cmd) {
    LCD_RS = 0;
    LCD_RW = 0;
    LCD_DATA = cmd;
    LCD_EN = 1;
    delay_ms(2);
    LCD_EN = 0;
}

void lcd_write_data(unsigned char dat) {
    LCD_RS = 1;
    LCD_RW = 0;
    LCD_DATA = dat;
    LCD_EN = 1;
    delay_ms(2);
    LCD_EN = 0;
}

void lcd_init() {
    delay_ms(15);
    lcd_write_cmd(0x38);   // 8位, 2行, 5x7点阵
    lcd_write_cmd(0x0C);   // 显示开, 光标关
    lcd_write_cmd(0x06);   // 写入后地址+1
    lcd_write_cmd(0x01);   // 清屏
    delay_ms(2);
}

void lcd_set_cursor(unsigned char row, unsigned char col) {
    unsigned char addr;
    if (row == 0)
        addr = 0x80 + col;
    else
        addr = 0xC0 + col;
    lcd_write_cmd(addr);
}

void lcd_print_string(unsigned char row, unsigned char col,
                      unsigned char *str) {
    lcd_set_cursor(row, col);
    while (*str) {
        lcd_write_data(*str++);
    }
}
```

## 八、ADC0809模数转换

```c
sbit ADC_START = P3^6;
sbit ADC_EOC   = P3^3;
sbit ADC_OE    = P3^7;
sbit ADC_CLK   = P3^4;

unsigned char adc_read(unsigned char channel) {
    // 选择通道
    P1 = (P1 & 0xF8) | channel;

    // 启动转换
    ADC_START = 0;
    ADC_START = 1;
    ADC_START = 0;

    // 等待转换完成
    while (ADC_EOC == 0);

    // 读取结果
    ADC_OE = 1;
    unsigned char result = P0;
    ADC_OE = 0;

    return result;
}
```
